JavaScript Integration

class GiveHubAPI { constructor(baseUrl) { this.baseUrl = baseUrl; this.token = null; this.csrfToken = null; } async init() { // Get CSRF token const csrfResponse = await fetch(`${this.baseUrl}/csrf_token.php`); const csrfData = await csrfResponse.json(); this.csrfToken = csrfData.token; } async login(username, password) { const response = await fetch(`${this.baseUrl}/api/auth/login`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-CSRF-Token': this.csrfToken }, body: JSON.stringify({ username, password }) }); const data = await response.json(); if (data.success) { this.token = data.tokens.accessToken; } return data; } async getCampaigns() { return await this.request('GET', '/api.php/Campaign'); } async createDonation(campaignId, amount, currency = 'XLM') { return await this.request('POST', '/blockchain-transaction-api.php', { campaign_id: campaignId, amount: amount, currency: currency }); } async request(method, endpoint, body = null) { const options = { method, headers: { 'Authorization': `Bearer ${this.token}`, 'Content-Type': 'application/json' } }; if (body && method !== 'GET') { options.body = JSON.stringify(body); options.headers['X-CSRF-Token'] = this.csrfToken; } const response = await fetch(`${this.baseUrl}${endpoint}`, options); return await response.json(); } } // Usage example const api = new GiveHubAPI('https://app.thegivehub.com'); await api.init(); await api.login('your-username', 'your-password'); const campaigns = await api.getCampaigns(); console.log('Available campaigns:', campaigns);

<?php class GiveHubAPI { private $baseUrl; private $token; private $csrfToken; public function __construct($baseUrl) { $this->baseUrl = rtrim($baseUrl, '/'); $this->initCsrf(); } private function initCsrf() { $response = file_get_contents($this->baseUrl . '/csrf_token.php'); $data = json_decode($response, true); $this->csrfToken = $data['token']; } public function login($username, $password) { $data = [ 'username' => $username, 'password' => $password ]; $options = [ 'http' => [ 'header' => [ "Content-Type: application/json", "X-CSRF-Token: {$this->csrfToken}" ], 'method' => 'POST', 'content' => json_encode($data) ] ]; $context = stream_context_create($options); $response = file_get_contents($this->baseUrl . '/api/auth/login', false, $context); $result = json_decode($response, true); if ($result['success']) { $this->token = $result['tokens']['accessToken']; } return $result; } public function getCampaigns() { return $this->request('GET', '/api.php/Campaign'); } public function createDonation($campaignId, $amount, $currency = 'XLM') { return $this->request('POST', '/blockchain-transaction-api.php', [ 'campaign_id' => $campaignId, 'amount' => $amount, 'currency' => $currency ]); } private function request($method, $endpoint, $data = null) { $headers = [ "Authorization: Bearer {$this->token}", "Content-Type: application/json" ]; if ($data && $method !== 'GET') { $headers[] = "X-CSRF-Token: {$this->csrfToken}"; } $options = [ 'http' => [ 'header' => implode("\r\n", $headers), 'method' => $method ] ]; if ($data && $method !== 'GET') { $options['http']['content'] = json_encode($data); } $context = stream_context_create($options); $response = file_get_contents($this->baseUrl . $endpoint, false, $context); return json_decode($response, true); } } // Usage example $api = new GiveHubAPI('https://app.thegivehub.com'); $loginResult = $api->login('your-username', 'your-password'); if ($loginResult['success']) { $campaigns = $api->getCampaigns(); echo "Available campaigns: " . json_encode($campaigns, JSON_PRETTY_PRINT); } ?>

import requests import json class GiveHubAPI: def __init__(self, base_url): self.base_url = base_url.rstrip('/') self.token = None self.csrf_token = None self.session = requests.Session() self._init_csrf() def _init_csrf(self): """Initialize CSRF token""" response = self.session.get(f"{self.base_url}/csrf_token.php") data = response.json() self.csrf_token = data['token'] def login(self, username, password): """Authenticate user and store JWT token""" data = { 'username': username, 'password': password } headers = { 'Content-Type': 'application/json', 'X-CSRF-Token': self.csrf_token } response = self.session.post( f"{self.base_url}/api/auth/login", json=data, headers=headers ) result = response.json() if result.get('success'): self.token = result['tokens']['accessToken'] return result def get_campaigns(self): """Get all campaigns""" return self._request('GET', '/api.php/Campaign') def create_donation(self, campaign_id, amount, currency='XLM'): """Create a donation transaction""" data = { 'campaign_id': campaign_id, 'amount': amount, 'currency': currency } return self._request('POST', '/blockchain-transaction-api.php', data) def _request(self, method, endpoint, data=None): """Make authenticated API request""" headers = { 'Authorization': f'Bearer {self.token}', 'Content-Type': 'application/json' } if data and method != 'GET': headers['X-CSRF-Token'] = self.csrf_token response = self.session.request( method, f"{self.base_url}{endpoint}", json=data if method != 'GET' else None, headers=headers ) return response.json() # Usage example api = GiveHubAPI('https://app.thegivehub.com') login_result = api.login('your-username', 'your-password') if login_result.get('success'): campaigns = api.get_campaigns() print("Available campaigns:", json.dumps(campaigns, indent=2)) # Create a donation donation_result = api.create_donation('campaign_123', 100, 'XLM') print("Donation result:", json.dumps(donation_result, indent=2))
